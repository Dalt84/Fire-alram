Introduction

This project uses an Arduino Uno, a DHT11 temperature sensor, a buzzer, RGB LEDs, and an LCD screen to provide fire alerts. When the temperature reaches a preset range, all alarm systems engage simultaneously: the buzzer sounds to alert those who are not hearing impaired, the RGB LEDs flash to visually warn the hearing impaired, and the LCD screen displays the current temperature along with the message Fire! Exit Now. This ensures comprehensive safety by combining auditory and visual signals.

Circuit Diagram

The ultrasonic sensor measures distance; the Arduino activates LEDs and the buzzer according to distance thresholds.

Environmental Impact Assessment

  • Materials: The Arduino Uno is RoHS and WEEE compliant, limiting hazardous substances.
  • Energy Efficiency: Circuit draw (standby): ~90–100 mA at 5 V (~0.45 W). Circuit draw (alarm active): ~150 mA at 5 V (~0.75 W). Overall efficiency: Minimal power consumption, suitable for USB supply or small solar panel integration.
  • Reusability: The fire alarm circuit demonstrates strong environmental responsibility: Low energy demand makes it compatible with sustainable power sources. Reusable modular components reduce long-term waste. Compliance with safety standards ensures minimal hazardous material use. With careful sourcing and efficient operation, the project aligns well with eco-friendly design principles..
  • Packaging & Shipping: Ordering all components from a single supplier helps minimize carbon footprint.

Arduino Code

The following Arduino code controls the system, lighting LEDs and activating the buzzer based on distance readings from the HC-SR04 sensor.

  
  
// --- LIBRARIES ---

#include "DHT.h"      // Library for the DHT sensor

#include  // Library for the LCD



// --- PIN DEFINITIONS ---

// Sensor Input

#define DHTPIN 13    // Digital pin for the DHT11 sensor

#define DHTTYPE DHT11  // Specifies the DHT11 sensor model



// Alarm Outputs

#define BUZZER_PIN 7  // Digital pin for the Buzzer

#define RED_LED_PIN 6  // Digital pin for the Single Red LED



// LCD Pins (Based on your input)

const int rs = 12, en = 11, d4 = 5, d5 = 4, d6 = 3, d7 = 2;

LiquidCrystal lcd(rs, en, d4, d5, d6, d7); 



// Initialize DHT sensor

DHT dht(DHTPIN, DHTTYPE);



// --- ALARM LOGIC VARIABLES ---

const float TEMP_THRESHOLD = 26.0;   // Absolute temperature limit (40 C)

const float TEMP_RISE_THRESHOLD = 0.5; // Temperature rise limit (5 C)

const unsigned long CHECK_INTERVAL = 5000; // Time interval for rate-of-change check (5000 ms = 5 seconds)



float lastTemp = 0.0;        // Stores the temperature reading from the last 5-second check

unsigned long lastTime = 0;     // Stores the time (ms) of the last check

bool alarmActive = false;      // Flag to track if the alarm is currently sounding



// --- SETUP FUNCTION: Runs once when the Arduino starts ---

void setup() {

 Serial.begin(9600); // Start serial communication for debugging and monitoring

 Serial.println("Fire Alarm System Initializing...");



 // Set up pins as outputs

 pinMode(BUZZER_PIN, OUTPUT);

 pinMode(RED_LED_PIN, OUTPUT);



 // Initialize DHT sensor

 dht.begin();



 // Initialize LCD (16 columns, 2 rows)

 lcd.begin(16, 2); 

 lcd.print("System Ready"); 

 lcd.setCursor(0, 1);

 lcd.print("Monitoring Temp");



 delay(2000);

}



// --- MAIN LOOP FUNCTION: Runs repeatedly forever ---

void loop() {

 // Read temperature (C) from DHT11

 float currentTemp = dht.readTemperature();



 // Check for sensor read failure

 if (isnan(currentTemp)) {

  Serial.println("Failed to read from DHT sensor!");

  delay(2000);

  return;

 }



 Serial.print("Current Temp (C): ");

 Serial.println(currentTemp);



 // --- ALARM CONDITION CHECK ---

 bool conditionMet = false;



 // 1. Check Absolute Temperature Threshold (40 C)

 if (currentTemp >= TEMP_THRESHOLD) {

  conditionMet = true;

  Serial.println("CONDITION 1: Absolute Temp Threshold MET (>= 40C)");

 }



 // 2. Check Rate of Change Threshold (5 C rise in 5 seconds)

 if (millis() - lastTime >= CHECK_INTERVAL) {

   

  // Only check rate of change after the first stable reading

  if (lastTemp != 0.0) {

   float tempDifference = currentTemp - lastTemp;



   if (tempDifference >= TEMP_RISE_THRESHOLD) {

    conditionMet = true;

    Serial.print("CONDITION 2: Rate of Change Threshold MET! Temp rise: ");

    Serial.print(tempDifference);

    Serial.println("C in 5s");

   }

  }



  // Update variables for the next 5-second check

  lastTemp = currentTemp;

  lastTime = millis();

 }



 // --- ALARM ACTIVATION / DEACTIVATION CONTROL ---

 if (conditionMet) {

  activateAlarms(currentTemp);

 } else {

  deactivateAlarms(currentTemp);

 }



 // Brief delay to prevent continuous, rapid loop execution

 delay(50); 

}



// --- ALARM FUNCTIONS ---



/**

 * Activates the Buzzer, Flashing Red LED, and LCD warning.

 */

void activateAlarms(float temp) {

 if (!alarmActive) {

  Serial.println("!!! ALARM ACTIVATED - FIRE !!!");

  alarmActive = true;

 }

  

 // 1. Buzzer (Audible Alert)

 // Continuous tone while alarm is active.

 tone(BUZZER_PIN, 1000); 



 // 2. Red LED (Visual Alert): Flash Red

 // The LED state alternates rapidly inside the loop

 digitalWrite(RED_LED_PIN, HIGH);

 delay(150);

 digitalWrite(RED_LED_PIN, LOW); // LED turns off briefly

 delay(150);



 // 3. LCD Screen: Display warning

 lcd.clear();

 lcd.print("FIRE! EXIT NOW");

 lcd.setCursor(0, 1);

 lcd.print("Temp: ");

 lcd.print(temp, 1); // Display temperature with one decimal place

 lcd.print("C");

}



/**

 * Returns the system to a normal, quiet state.

 */

void deactivateAlarms(float temp) {

 if (alarmActive) {

  Serial.println("Alarm Deactivated. System Normal.");

  alarmActive = false;

   

  // Stop all alarm outputs

  noTone(BUZZER_PIN); 

  digitalWrite(RED_LED_PIN, LOW); 



  // Update LCD to System Normal message

  lcd.clear();

  lcd.print("System Normal");

  lcd.setCursor(0, 1);

  lcd.print("Temp: ");

  lcd.print(temp, 1);

  lcd.print("C");

 }

  

 // Update the temperature on the LCD if the system is already normal

 if (!alarmActive) {

  lcd.setCursor(6, 1); // Move cursor to where the temperature starts (after "Temp: ")

  lcd.print(temp, 1);

  lcd.print("C  "); // Print spaces to clear old digits

 }

}

Conclusion

The fire alarm circuit project demonstrates a balanced approach to safety and sustainability. By utilizing RoHS- and WEEE-compliant components, the design minimizes hazardous material use. Its low energy demand—less than 1 W even under alarm conditions—makes it suitable for efficient power sources such as USB or solar. The modular nature of the system ensures components can be reused, extending their lifecycle and reducing electronic waste. With thoughtful sourcing and consolidated shipping, the project further reduces its carbon footprint. Overall, the fire alarm circuit provides essential safety functionality while aligning with environmentally responsible practices, making it a model for small-scale, eco-conscious electronic design.
